80 cipherTensor pTensor::binaryOpAbstraction(const char *flag,
81  pTensor other) {
82  cipherTensor ciphertextContainer;
83 
84  // Now, we know that it is broadcast-able. Therefore, they either have the same shape or one is shape 1 in rows
85  cipherVector op_res;
86  int ringDim = (*m_cc)->GetRingDimension();
87  for (unsigned int i = 0; i < std::max(m_rows, other.m_rows); i++) {
88  unsigned int lhsInd;
89  unsigned int rhsInd;
90 
91  if (m_rows == other.m_rows) {
92  lhsInd = i;
93  rhsInd = i;
94  } else if (m_rows > other.m_rows) {
95  lhsInd = i;
96  rhsInd = 0;
97  } else {
98  lhsInd = 0;
99  rhsInd = i;
100  }
101  if (other.m_isEncrypted) {
102  cipherVector otherVec;
103 
104  // If it is a scalar that has not been repeated, we repeat it.
105  if (other.isScalar() && !(other.m_isRepeated)) {
106 
107  // Get how much to sum over and rotate.
108  int rot = int(-ringDim / 4) + 1;
109  // We've now summed it up and it should be projected into the back
110  otherVec = (*m_cc)->EvalSum(other.m_ciphertexts[0], -rot);
111  // The last rot entries are now populated with the value. We then rotate them back and we are done.
112  otherVec = (*m_cc)->EvalAtIndex(otherVec, rot);
113 
114  } else {
115  otherVec = other.m_ciphertexts[rhsInd];
116  }
117  op_res = applyBinaryOp(flag, m_ciphertexts[lhsInd], otherVec);
118  } else {
119  messageVector otherVec;
120  if (other.isScalar()) { // We repeat the value n_cols times across then do the operation.
121  otherVec = messageVector(m_cols, other.m_messages[0][0]);
122  } else {
123  otherVec = other.m_messages[rhsInd];
124  }
125  lbcrypto::Plaintext intermediary = (*m_cc)->MakeCKKSPackedPlaintext(otherVec);
126  op_res = applyBinaryOp(flag, m_ciphertexts[lhsInd], intermediary);
127  }
128  ciphertextContainer.emplace_back(op_res);
129  }
130  return ciphertextContainer;
131 }

273

378 pTensor pTensor::T() {
379  // For the first row we iteratively mask out
380  if (!(m_isEncrypted)) { // encrypt yourself. If we store the transpose we have it now in encrypted form.
381  pTensor::encrypt();
382  }
383 
384  // At this point, our m_TCiphertexts is either empty or not empty. If not empty, the user either didn't
385  // want to encrypt it from the get-go OR we were passed in a ciphertext directly. In that case, we do
386  // the slow encrypted transpose.
387  // If NOT empty, we have the transpose that we return.
388  if (!m_TCiphertexts.empty()) {
389  // we flip the rows and cols. We then reassign
390  pTensor newTensor(m_cols, m_rows, m_TCiphertexts, m_ciphertexts);
391  return newTensor;
392  }
393  // Whelp, we need to transpose here I suppose.
394  // We fix a column then iterate downwards through the samples. At the end, we have a vector
395  // at which point we emplace back then move to the next col
396  auto toTranspose = (m_ciphertexts);
397 
398  cipherTensor tContainer;
399  for (unsigned int col_i = 0; col_i < m_cols; ++col_i) {
400  messageVector mAccum(m_rows, 0.0);
401  auto accum = (*m_cc)->Encrypt(m_public_key, (*m_cc)->MakeCKKSPackedPlaintext(mAccum));
402  for (unsigned int row_i = 0; row_i < m_rows; ++row_i) {
403  messageVector mask(m_rows, 0.0);
404  mask[row_i] = 1;
405  auto ptMask = (*m_cc)->MakeCKKSPackedPlaintext(mask);
406 
407  // First mask everything else out
408  auto maskedVal = (*m_cc)->EvalMult(ptMask, toTranspose[col_i]);
409 
410  // Now, we rotate right: -1
411  for (unsigned int rot = 0; rot < row_i; ++rot) {
412  maskedVal = (*m_cc)->EvalAtIndex(maskedVal, 1);
413  }
414  accum = (*m_cc)->EvalAdd(accum, maskedVal);
415  }
416  tContainer.emplace_back(accum);
417  }
418  pTensor newTensor(m_cols, m_rows, tContainer, toTranspose);
419  return newTensor;
420 }
